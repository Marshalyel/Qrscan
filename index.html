// index.js (Electron Main Process) const { app, BrowserWindow, session } = require('electron'); const path = require('path'); function createWindow() { // Buat window browser baru const win = new BrowserWindow({ width: 480, height: 800, webPreferences: { // kalau kamu butuh akses node di renderer, set ini true: nodeIntegration: false, contextIsolation: true, } }); // Load file index.html dari folder yang sama win.loadFile(path.join(__dirname, 'index.html')); // Buka DevTools (hapus baris ini saat produksi) win.webContents.openDevTools({ mode: 'detach' }); } // Tunggu hingga Electron siap app.whenReady() .then(() => { // (Opsional) biar getUserMedia di Electron jalan tanpa permission popup session.defaultSession.setPermissionRequestHandler((webContents, permission, callback) => { if (permission === 'media') { return callback(true); } callback(false); }); createWindow(); // Re‑open window di macOS saat dock icon di‑klik app.on('activate', () => { if (BrowserWindow.getAllWindows().length === 0) { createWindow(); } }); }); // Quit app saat semua window ditutup (kecuali di macOS) app.on('window-all-closed', () => { if (process.platform !== 'darwin') { app.quit(); } });
